import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class AccueilReservationParc extends JFrame {

    public AccueilReservationParc() {
        setTitle("🎟️ Parc Aventure - Réservation");
        setSize(500, 350);
        setLocationRelativeTo(null);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setResizable(false);

        JPanel mainPanel = new JPanel(new BorderLayout());
        mainPanel.setBackground(new Color(255, 245, 204)); // jaune pâle

        // Titre
        JLabel titre = new JLabel("🎡 Bienvenue au Parc Aventure 🎢", SwingConstants.CENTER);
        titre.setFont(new Font("Arial", Font.BOLD, 22));
        titre.setBorder(BorderFactory.createEmptyBorder(20, 0, 10, 0));
        mainPanel.add(titre, BorderLayout.NORTH);

        // Zone centrale (peut être utilisée pour afficher une image ou description)
        JTextArea description = new JTextArea(
            "Réservez vos billets et découvrez nos attractions spectaculaires !\n" +
            "Profitez d'une journée inoubliable en famille ou entre amis.");
        description.setEditable(false);
        description.setWrapStyleWord(true);
        description.setLineWrap(true);
        description.setFont(new Font("Serif", Font.PLAIN, 16));
        description.setBackground(new Color(255, 245, 204));
        description.setMargin(new Insets(10, 20, 10, 20));
        mainPanel.add(description, BorderLayout.CENTER);

        // Boutons en bas
        JPanel boutonPanel = new JPanel(new FlowLayout());
        boutonPanel.setBackground(new Color(255, 245, 204));

        JButton reserverBtn = new JButton("🎫 Réserver un billet");
        JButton attractionsBtn = new JButton("🎠 Voir les attractions");
        JButton quitterBtn = new JButton("❌ Quitter");

        reserverBtn.addActionListener(e ->
                JOptionPane.showMessageDialog(this, "Interface de réservation à venir.")
        );

        attractionsBtn.addActionListener(e ->
                JOptionPane.showMessageDialog(this, "Liste des attractions à venir.")
        );

        quitterBtn.addActionListener(e -> System.exit(0));

        boutonPanel.add(reserverBtn);
        boutonPanel.add(attractionsBtn);
        boutonPanel.add(quitterBtn);

        mainPanel.add(boutonPanel, BorderLayout.SOUTH);

        add(mainPanel);
        setVisible(true);
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(AccueilReservationParc::new);
    }
}

  public class AccueilMinimaliste extends JFrame {

      public AccueilMinimaliste() {
          setTitle("Parc Attraction - Accueil");
          setSize(400, 300);
          setLocationRelativeTo(null);
          setDefaultCloseOperation(EXIT_ON_CLOSE);
          setResizable(false);

          JPanel panel = new JPanel();
          panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
          panel.setBackground(Color.WHITE);

          JLabel titre = new JLabel("🎢 Parc d'Attractions");
          titre.setAlignmentX(Component.CENTER_ALIGNMENT);
          titre.setFont(new Font("SansSerif", Font.BOLD, 22));
          titre.setBorder(BorderFactory.createEmptyBorder(20, 0, 10, 0));

          JLabel sousTitre = new JLabel("Réservez votre aventure !");
          sousTitre.setAlignmentX(Component.CENTER_ALIGNMENT);
          sousTitre.setFont(new Font("SansSerif", Font.PLAIN, 14));
          sousTitre.setBorder(BorderFactory.createEmptyBorder(0, 0, 20, 0));

          JButton btnReserver = new JButton("Réserver un billet");
          JButton btnVoir = new JButton("Voir les attractions");
          JButton btnQuitter = new JButton("Quitter");

          btnReserver.setAlignmentX(Component.CENTER_ALIGNMENT);
          btnVoir.setAlignmentX(Component.CENTER_ALIGNMENT);
          btnQuitter.setAlignmentX(Component.CENTER_ALIGNMENT);

          btnQuitter.addActionListener(e -> System.exit(0));

          panel.add(titre);
          panel.add(sousTitre);
          panel.add(btnReserver);
          panel.add(Box.createVerticalStrut(10));
          panel.add(btnVoir);
          panel.add(Box.createVerticalStrut(10));
          panel.add(btnQuitter);

          add(panel);
          setVisible(true);
      }

      public static void main(String[] args) {
          SwingUtilities.invokeLater(AccueilMinimaliste::new);
      }
  }
